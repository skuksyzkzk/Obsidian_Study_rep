이렇게 같은 Player 컴포넌트 이지만 
두개는 분리되어서 내가 Player 1의 버튼을 클릭하면 2에는 영향이안간다 
```jsx
import { useState } from "react"

export default function Player({ name, symbol }) {

    const [isEditing, setIsEditing] = useState(false);

    // false라는 것은 Edit 할 마음이없다는 것으로 span을 보여준다 true일 경우에는 input을 띄운다

    let nameLabel = <span className="player-name">{name}</span>;

    if (isEditing) {

        nameLabel = <input type="text"></input>

    }

  

    function handleEditing() {

        if (!isEditing) {

            setIsEditing(true);

        } else {

            setIsEditing(false);

        }

    }

    return (

        <li>

            <span className="player">

                {nameLabel}

                <span className="player-symbol">{symbol}</span>

            </span>

            <button onClick={handleEditing}>Edit</button>

        </li>

    )

}
```
